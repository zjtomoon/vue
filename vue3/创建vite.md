+ 安装命令行工具

```bash
yarn global add @vue/cli
# 或
npm install -g @vue/cli
```

+ 构建工具vite

```bash
# npm 6.x
$ npm init vite@latest <project-name> --template vue

# npm 7+，需要加上额外的双短横线
$ npm init vite@latest <project-name> -- --template vue

$ cd <project-name>
$ npm install
$ npm run dev


$ yarn create vite <project-name> --template vue
$ cd <project-name>
$ yarn
$ yarn dev

$ pnpm create vite <project-name> -- --template vue
$ cd <project-name>
$ pnpm install
$ pnpm dev
```
